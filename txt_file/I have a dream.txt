 /**
         *
         * @param str 判断中英文
         * @returns {boolean} 中文返回true，其他返回false
         */
        function isChinese(str) {
            var entryVal = str;
            var cnChar = entryVal.match(/[^\x00-\x80]/g);
            if (cnChar != null && cnChar.length > 0)
                return true;
            return false;
        }
